\section{Mobile Device Hygiene}
\label{sec:mobile}

\subsection{App Audit}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Delete unused apps. & $\square$ Clear app caches where necessary. \\ 
$\square$ Review app permissions (camera, microphone, location).  & \\ 
\end{tabular}

\subsection{Storage and Media}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Remove duplicate photos/videos. & $\square$ Backup photos to secure storage. \\ 
$\square$ Delete sensitive screenshots or documents. & \\ 
\end{tabular}

\subsection{System Updates}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Update OS and apps. & $\square$ Enable automatic updates. \\ 
$\square$ Restart device after updates. & \\ 
\end{tabular}
