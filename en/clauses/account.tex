\section{Account and Password Security}
\label{sec:account}

\subsection{Audit Online Accounts}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ List all active accounts (email, banking, subscriptions, social media). & $\square$ Close unused or duplicate accounts. \\ 
$\square$ Remove saved payment methods where not needed. & \\ 
\end{tabular}

\subsection{Strengthen Passwords}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Use a password manager. & $\square$ Replace weak/reused passwords. \\ 
$\square$ Enable multi-factor authentication (MFA) on all critical accounts. & $\square$ Update recovery email and phone number. \\ 
\end{tabular}

\subsection{Monitor Breaches}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Check for data breaches. & $\square$ Change passwords for compromised accounts immediately. \\ 
\end{tabular}
